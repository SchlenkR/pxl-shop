MyClockFace.cs 
public class MyClockFace : PxlApp
{
    public override void Render(PxlCanvas canvas)
    {
        // Get current time
        var now = DateTime.Now;

        // Draw colorful background
        canvas.DrawGradient(
            Color.Cyan,
            Color.Magenta
        );

        // Display time in pixels
        canvas.DrawText(
            now.ToString("HH:mm"),
            Color.White
        );
    }
}
